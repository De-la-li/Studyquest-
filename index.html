🗄️ Supabase Setup Guide
Run these SQL queries in Supabase → SQL Editor → New Query. Do them in order.
1
Users Table
CREATE TABLE users (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  email TEXT UNIQUE NOT NULL,
  first_name TEXT NOT NULL,
  last_name TEXT NOT NULL,
  school TEXT, program TEXT, year TEXT,
  plan TEXT DEFAULT 'free',
  xp INTEGER DEFAULT 0,
  streak INTEGER DEFAULT 0,
  best_streak INTEGER DEFAULT 0,
  last_checkin DATE, pay_ref TEXT,
  plan_start TIMESTAMPTZ,
  joined_at TIMESTAMPTZ DEFAULT NOW(),
  is_admin BOOLEAN DEFAULT FALSE
);
2
Courses Table
CREATE TABLE courses (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  code TEXT, name TEXT NOT NULL,
  lecturer TEXT, venue TEXT,
  days TEXT[], start_time TEXT, end_time TEXT,
  credits INTEGER, color TEXT DEFAULT '#7c5cfc',
  created_at TIMESTAMPTZ DEFAULT NOW()
);
3
Messages Table
CREATE TABLE messages (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  role TEXT NOT NULL, content TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
4
Feedback Table
CREATE TABLE feedback (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES users(id) ON DELETE SET NULL,
  type TEXT NOT NULL, title TEXT,
  body TEXT NOT NULL, stars INTEGER,
  anon BOOLEAN DEFAULT FALSE,
  votes INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
5
Quiz Stats Table
CREATE TABLE quiz_stats (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  topic TEXT, score INTEGER, total INTEGER,
  xp_earned INTEGER, best_streak INTEGER, mode TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
6
Enable Row Level Security
ALTER TABLE users      ENABLE ROW LEVEL SECURITY;
ALTER TABLE courses    ENABLE ROW LEVEL SECURITY;
ALTER TABLE messages   ENABLE ROW LEVEL SECURITY;
ALTER TABLE feedback   ENABLE ROW LEVEL SECURITY;
ALTER TABLE quiz_stats ENABLE ROW LEVEL SECURITY;

CREATE POLICY "users_own" ON users
  FOR ALL USING (auth.uid()::text = id::text);
CREATE POLICY "courses_own" ON courses
  FOR ALL USING (auth.uid()::text = user_id::text);
CREATE POLICY "messages_own" ON messages
  FOR ALL USING (auth.uid()::text = user_id::text);
CREATE POLICY "feedback_read" ON feedback
  FOR SELECT USING (true);
CREATE POLICY "feedback_write" ON feedback
  FOR INSERT WITH CHECK (auth.uid()::text = user_id::text);
7
Make yourself admin (run after signup)
-- Replace with YOUR email
UPDATE users SET plan = 'pro', is_admin = TRUE
WHERE email = 'YOUR_EMAIL_HERE';
✅ After all queries run
Go to Supabase → Settings → API → copy your Project URL and anon public key. Add them to Netlify environment variables.
